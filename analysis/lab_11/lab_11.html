states_map <- map_data("state")

deaths_map <- ds_combine %>% 
  filter(year == 2015) %>% 
  mutate(state = tolower(state)) %>% 
  select(state,drug_overdose_death_rate, motor_death_rate) %>% 
  merge(states_map, by.x = "state", by.y = "region")

deaths_map_long <- deaths_map %>%

  mutate(
    death_type = factor(
